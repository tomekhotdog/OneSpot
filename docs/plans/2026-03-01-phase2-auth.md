```python
import logging
from backend import config
from backend.models import WhatsAppLogEntry
from backend.state import StateManager, state_manager as default_sm

logger = logging.getLogger(__name__)


def send_whatsapp_message(
    recipient: str,
    template: str,
    params: dict,
    state_manager: StateManager | None = None,
) -> dict:
    sm = state_manager or default_sm

    if config.WHATSAPP_MOCK:
        logger.info(f"[MOCK WhatsApp] To: {recipient} Template: {template} Params: {params}")
        entry = WhatsAppLogEntry(
            recipient=recipient, template=template, params=params, status="sent"
        )
        def log_message(state):
            state.whatsapp_log.append(entry)
            return state
        sm.update(log_message)
        return {"status": "sent", "mock": True, "id": entry.id}

    # Real WhatsApp API — to be implemented in Phase 3
    raise NotImplementedError("Real WhatsApp API not yet implemented")
```

**Step 2: Implement otp.py**

```python
import hashlib
import hmac
import secrets
from datetime import datetime, timedelta

from backend import config
from backend.models import OTPRequest
from backend.state import StateManager, state_manager as default_sm


class OTPError(Exception):
    pass


def generate_otp(phone: str, state_manager: StateManager | None = None) -> dict:
    sm = state_manager or default_sm
    state = sm.read()
    now = datetime.utcnow()

    existing = state.otp_requests.get(phone)
    if existing:
        window_start = existing.window_start
        if (now - window_start).total_seconds() < config.OTP_RATE_LIMIT_WINDOW_SECONDS:
            if existing.request_count_window >= config.OTP_RATE_LIMIT_MAX_REQUESTS:
                raise OTPError("Too many OTP requests — rate limit exceeded")
            count = existing.request_count_window + 1
        else:
            window_start = now
            count = 1
    else:
        window_start = now
        count = 1

    code = f"{secrets.randbelow(1000000):06d}"
    otp = OTPRequest(
        code=code,
        created_at=now,
        expires_at=now + timedelta(seconds=config.OTP_EXPIRY_SECONDS),
        attempts=0,
        request_count_window=count,
        window_start=window_start,
    )

    def store(state):
        state.otp_requests[phone] = otp
        return state

    sm.update(store)
    return {"code": code, "expires_in": config.OTP_EXPIRY_SECONDS}


def verify_otp(phone: str, code: str, state_manager: StateManager | None = None) -> bool:
    sm = state_manager or default_sm
    state = sm.read()
    otp = state.otp_requests.get(phone)

    if not otp:
        raise OTPError("No OTP requested for this number")

    now = datetime.utcnow()
    if now > otp.expires_at:
        raise OTPError("OTP has expired")

    if otp.attempts >= config.OTP_MAX_ATTEMPTS:
        raise OTPError("Too many failed attempts — max attempts reached")

    if otp.code != code:
        def inc_attempts(state):
            state.otp_requests[phone].attempts += 1
            return state
        sm.update(inc_attempts)
        return False

    # Correct code — remove OTP from state
    def remove_otp(state):
        del state.otp_requests[phone]
        return state
    sm.update(remove_otp)
    return True
```
