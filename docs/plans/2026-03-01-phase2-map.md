```jsx
import { bayStatusColors } from '../styles/theme'

const statusStyles = {
  available: 'bg-primary text-white shadow-md cursor-pointer hover:bg-primary-dark',
  own: 'bg-accent-green text-white',
  unavailable: 'bg-gray-300 text-gray-500',
  booked: 'bg-accent-amber text-white',
  restricted: 'bg-gray-200 text-gray-400',
}

export default function BayCell({ bay, onSelect }) {
  const style = statusStyles[bay.status] || statusStyles.unavailable

  return (
    <button
      onClick={() => bay.status === 'available' && onSelect?.(bay)}
      disabled={bay.status !== 'available'}
      className={`w-12 h-12 rounded-lg flex items-center justify-center text-xs font-semibold transition-all ${style}`}
      title={`Bay ${bay.number} — ${bay.status}`}
    >
      {bay.number.split('-')[1]}
    </button>
  )
}
```

### Task 3: ParkingMap Component

**Files:**
- Create: `frontend/src/components/ParkingMap.jsx`

**Step 1: Implement ParkingMap.jsx**

Renders one level at a time. Tab switcher for Level A / Level B. Grid layout based on bay row/col positions from the API. Each cell is a BayCell.

```jsx
import { useState } from 'react'
import BayCell from './BayCell'

export default function ParkingMap({ bays, levels, onSelectBay }) {
  const [activeLevel, setActiveLevel] = useState(levels[0]?.id || 'A')

  const levelBays = bays.filter(b => b.level === activeLevel)
  const level = levels.find(l => l.id === activeLevel)
  const rows = level?.rows || 5
  const cols = level?.cols || 15

  // Build grid
  const grid = Array.from({ length: rows }, () => Array(cols).fill(null))
  levelBays.forEach(bay => {
    if (bay.row < rows && bay.col < cols) {
      grid[bay.row][bay.col] = bay
    }
  })

  return (
    <div>
      {/* Level tabs */}
      <div className="flex gap-2 mb-4">
        {levels.map(l => (
          <button
            key={l.id}
            onClick={() => setActiveLevel(l.id)}
            className={`px-4 py-2 rounded-button font-semibold text-sm transition-colors ${
              activeLevel === l.id
                ? 'bg-primary text-white'
                : 'bg-bg-card text-text-secondary border border-border'
            }`}
          >
            {l.name}
          </button>
        ))}
      </div>

      {/* Bay grid */}
      <div
        className="grid gap-1 overflow-x-auto pb-2"
        style={{ gridTemplateColumns: `repeat(${cols}, 48px)` }}
      >
        {grid.flat().map((bay, i) =>
          bay ? (
            <BayCell key={bay.id} bay={bay} onSelect={onSelectBay} />
          ) : (
            <div key={`empty-${i}`} className="w-12 h-12" />
          )
        )}
      </div>

      {/* Legend */}
      <div className="flex flex-wrap gap-3 mt-4 text-xs text-text-secondary">
        <span className="flex items-center gap-1"><span className="w-3 h-3 rounded bg-primary" /> Available</span>
        <span className="flex items-center gap-1"><span className="w-3 h-3 rounded bg-accent-green" /> Your bay</span>
        <span className="flex items-center gap-1"><span className="w-3 h-3 rounded bg-accent-amber" /> Booked</span>
        <span className="flex items-center gap-1"><span className="w-3 h-3 rounded bg-gray-300" /> Unavailable</span>
      </div>
    </div>
  )
}
```
